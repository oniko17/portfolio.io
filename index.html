    menu.addEventListener("click", (e) => {
        if (e.target.matches("._menu-ul a")) {
            e.preventDefault();
            
            const id = e.target.getAttribute("href").substring(1);

            const pages = document.querySelectorAll(".page");
            pages.forEach((page) => {
                page.classList.add("hidePage");
            });

        const targetPage = document.getElementById(id);
        targetPage.classList.remove("hidePage");
        }
        });

Привет! Меня зовут Илья Звягинцев . Я программист, и я просто обожаю кодить! Раньше я занимался программированием на любительском уровне для собственного удовольствия, но потом принял решение сменить профессию и погрузиться в это дело более глубоко.

Для того чтобы освоить все тонкости и принципы программирования, я прошел курсы в GeekBrains по направлению программирование и адаптивная верстка. Во время обучения я освоил язык программирования C# на базовом уровне, а также Python . Кроме того, я изучил HTML, CSS и JavaScript , а также немного познакомился с библиотекой React .

Я всегда стремлюсь к самосовершенствованию и постоянно ищу новые знания. Недавно я успешно завершил обучение по направлению 1С: Битрикс - управление сайтом , контент-менеджмент и администрирование на базовом уровне. Это дало мне возможность приобрести практические навыки в разработке и управлении сайтами, используя этот популярный инструмент.

Я считаю, что программирование - это искусство, которое позволяет мне творить и воплощать свои идеи в жизнь. Я увлечен созданием эффективных и интуитивно понятных решений, которые помогут улучшить жизнь людей и деловые процессы.
